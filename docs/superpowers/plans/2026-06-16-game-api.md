# 弈仙牌 game-api 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把已逆向验证的游戏函数封装成 C# 面板层(`YiXianApi.dll`)+ Python 包装层(`yixian_api`),声明式注册表驱动,逐个活体验证,出 API 文档,供二次开发调用。

**Architecture:** C# 顶层静态类(`BattleApi`/`SceneApi`/`GameStatusApi`/`NetworkApi`/`StateApi`,namespace `YiXianBot`)直接调游戏 `BattleManager.Instance.*` / `AccountUtil.*`,编译成独立 `YiXianApi.dll`;Python 侧一张声明式**注册表**(`registry.py`)列出每个 API 的 `(namespace, name, c#类型, c#方法, 调用类型, 返回类型, 文档)`,`Client`/`validate.py`/`gen_docs.py` 全部遍历同一张表 —— 加一个新 API = C# 加一个方法 + 注册表加一行,文档与验证自动带出。

**Tech Stack:** C# net40 + ILRuntime 热更(`dotnet build`,引用 `native_hud/_refs/*.dll`);frida-python + `bot_glue3.agent.js` RPC(`load_bot`/`call_s`/`call_str`);Python 3 + pytest 7.4。

**关键约定:**
- **类名稳定不迭代**:`YiXianBot.BattleApi` 等类名固定(不学 Hud 每次改号)。`validate.py` 每次 **spawn 全新游戏**、全新 AppDomain 首次 `load_bot`,无 ILRuntime 同名去重问题。`AssemblyName=YiXianApi`。(仅当要往运行中的游戏热重载 API 时才需临时改 AssemblyName。)
- **RPC Python 侧方法名**(frida 把 JS `callS`→`call_s`):`ex.load_bot(bytes)`、`ex.call_s(type, method, [int...])`、`ex.call_str(type, method, "str")`、`ex.call_private(type, method)`。
- **C# 面板每个方法 try/catch → 返回 `"ok:..."` / `"EX:msg"` / JSON 字符串**,绝不跨 RPC 抛异常。
- 游戏类型(`BattleManager`/`AccountUtil`/`SceneType`/`GameStatus`)在 DarkSun.HotUpdate 全局命名空间,`namespace YiXianBot` 内可直接引用(与 `Hud.cs` 同)。

```csharp
    public static class BattleApi
    {
        // 任一执行器在跑 = 战斗动画进行中。
        public static string IsBattling()
        {
            try {
                var bm = BattleManager.Instance;
                if (bm == null || bm.allBattleExecuters == null) return ApiUtil.Ok("0");
                var list = bm.allBattleExecuters;
                for (int i = 0; i < list.Count; i++)
                { var be = list[i]; if (be != null && be.isExecuting) return ApiUtil.Ok("1"); }
                return ApiUtil.Ok("0");
            } catch (Exception e) { return ApiUtil.Ex(e); }
        }

        // 仅强制打断,不切场景(底层动作,供组合)。
        public static string ForceBreak()
        {
            try {
                var bm = BattleManager.Instance;
                if (bm == null || bm.allBattleExecuters == null) return ApiUtil.Ok("nobm");
                var list = bm.allBattleExecuters;
                int n = 0;
                for (int i = 0; i < list.Count; i++)
                { var be = list[i]; if (be != null && be.isExecuting) { be.forceBreakExecuting = true; n++; } }
                return ApiUtil.Ok("broke:" + n);
            } catch (Exception e) { return ApiUtil.Ex(e); }
        }

        // 完整跳过链路 = SkipBattleResultPanel.OnHide:100倍速冲到打断点 → 等全部
        // !isExecuting → ChangeSceneType(修炼阶段) → GameStatusReq()。用 EveryUpdate
        // 自轮询完成(不依赖 UniTask),立即返回 dispatched。
        static IDisposable s_skipSub;
        public static string Skip()
        {
            try {
                var bm = BattleManager.Instance;
                if (bm == null) return "EX:no BattleManager";
                if (bm.currentScene != SceneType.斗法阶段) return ApiUtil.Ok("noop:not-in-battle");
                Time.timeScale = 100f;
                if (bm.allBattleExecuters != null)
                {
                    var list = bm.allBattleExecuters;
                    for (int i = 0; i < list.Count; i++)
                    { var be = list[i]; if (be != null && be.isExecuting) be.forceBreakExecuting = true; }
                }
                if (s_skipSub != null) { s_skipSub.Dispose(); s_skipSub = null; }
                s_skipSub = ObservableExtensions.Subscribe(Observable.EveryUpdate(), new Action<long>(PumpSkip));
                return ApiUtil.Ok("dispatched");
            } catch (Exception e) { return ApiUtil.Ex(e); }
        }
        static void PumpSkip(long _)
        {
            try {
                var bm = BattleManager.Instance;
                if (bm == null) { StopSkip(); return; }
                if (bm.allBattleExecuters != null)
                {
                    var list = bm.allBattleExecuters;
                    for (int i = 0; i < list.Count; i++)
                    { if (list[i] != null && list[i].isExecuting) { Time.timeScale = 100f; return; } }
                }
                if (bm.currentScene == SceneType.斗法阶段)
                {
                    bm.ChangeSceneType(SceneType.修炼阶段);
                    if (!bm.isSpectating) bm.GameStatusReq();
                }
                Time.timeScale = 1f;
                StopSkip();
            } catch (Exception) { StopSkip(); }
        }
        static void StopSkip()
        { if (s_skipSub != null) { s_skipSub.Dispose(); s_skipSub = null; } }
    }
```

```csharp
    {
        // 当前对局轮次。
        public static string Round()
        {
            try {
                var bm = BattleManager.Instance;
                var gs = bm != null ? bm.currentGameStatus : null;
                if (gs == null) return "EX:no GameStatus";
                return "{\"round\":" + gs.round + "}";

        // 自身上一轮快照:命/体魄/上限/等级(JSON)。

        {
            try {

                var gs = bm != null ? bm.currentGameStatus : null;
                if (gs == null) return "EX:no GameStatus";
                var p = gs.GetMainPlayerData();

                  .Append(",\"maxHp\":").Append(d.maxHp)
                  .Append(",\"tiPo\":").Append(d.tiPo)
                  .Append(",\"level\":").Append(d.level).Append("}");
                return sb.ToString();
            } catch (Exception e) { return ApiUtil.Ex(e); }
        }
    }
```

```python
# -*- coding: utf-8 -*-
"""单一事实源:每个对外 API 一条声明。Client / validate / gen_docs 全部遍历这张表。
加一个新 API = C# 面板加一个方法 + 这里加一行 ApiSpec(文档与验证自动带出)。"""
from dataclasses import dataclass, field
from typing import List


@dataclass(frozen=True)
class ApiSpec:
    namespace: str          # python 子命名空间 (battle/scene/...)
    name: str               # python 方法名 (skip/auto_select/...)
    ctype: str              # C# 类型全名 (YiXianBot.BattleApi)
    cmethod: str            # C# 方法名 (Skip)
    call: str               # "call_s"(int 参数) | "call_str"(字符串参数)
    ret: str                # "status"(ok:/EX:) | "json"(读盘类)
    args: List[str] = field(default_factory=list)   # 参数名(顺序);call_s→int, call_str→str
    doc: str = ""           # 一行说明
    state_dep: str = ""     # 活体验证的场景依赖,如 "斗法阶段";空=任意场景可测


T_BATTLE = "YiXianBot.BattleApi"
T_SCENE = "YiXianBot.SceneApi"
T_GS = "YiXianBot.GameStatusApi"
T_NET = "YiXianBot.NetworkApi"
T_STATE = "YiXianBot.StateApi"

API: List[ApiSpec] = [
    ApiSpec("battle", "skip", T_BATTLE, "Skip", "call_s", "status", [],
            "跳过当前战斗动画,正常回到摆牌(发牌+加换牌次数)", "斗法阶段"),
    ApiSpec("battle", "force_break", T_BATTLE, "ForceBreak", "call_s", "status", [],
            "强制打断所有执行中的战斗执行器(不切场景)", "斗法阶段"),
    ApiSpec("battle", "is_battling", T_BATTLE, "IsBattling", "call_s", "status", [],
            "当前是否有战斗动画在播放(ok:1/ok:0)"),
    ApiSpec("scene", "change", T_SCENE, "Change", "call_s", "status", ["scene_type"],
            "切换场景:0=修炼阶段(摆牌) 1=斗法阶段(战斗)"),
    ApiSpec("scene", "current", T_SCENE, "Current", "call_s", "status", [],
            "当前场景枚举 int(0/1)"),
    ApiSpec("game_status", "req", T_GS, "Req", "call_s", "status", [],
            "向服务器请求当前权威对局状态(补发牌/换牌次数)"),
    ApiSpec("network", "auto_select", T_NET, "AutoSelect", "call_s", "status", ["ping_amount"],
            "自动选择最优线路(=设置-网络-自动选择)"),
```

```python
# -*- coding: utf-8 -*-
"""底层管道:连游戏(spawn/attach)、加载 YiXianApi.dll、等 AppDomain 就绪、调 RPC。
原始 call_s/call_str 把 RPC 返回 {ok,result,err} 解开成字符串,EX:/not found 抛错。"""
import os
import time
from pathlib import Path

REPO = Path(__file__).resolve().parents[2]
DLL = Path(os.environ.get("YX_API_DLL", REPO / "native_hud" / "_build" / "YiXianApi.dll"))
GLUE = Path(os.environ.get("YX_HUD_BUILD", REPO / "native_hud" / "_build")) / "bot_glue3.agent.js"
GAME = os.environ.get("YX_GAME_EXE", r"F:\SteamLibrary\steamapps\common\YiXianPai\YiXianPai.exe")
PROC = os.environ.get("YX_PROC", "YiXianPai.exe")


class YiXianApiError(Exception):
    def __init__(self, ctype, cmethod, raw):
        super().__init__("%s.%s -> %s" % (ctype, cmethod, raw))
        self.ctype, self.cmethod, self.raw = ctype, cmethod, raw


def _unwrap(ctype, cmethod, r):
    """{ok,result,err} → result 字符串;失败抛 YiXianApiError。"""
    if not r or not r.get("ok"):
        raise YiXianApiError(ctype, cmethod, (r or {}).get("err", "rpc-failed"))
    res = r.get("result", "")
    if isinstance(res, str) and (res.startswith("EX:") or res.startswith("not found")):
        raise YiXianApiError(ctype, cmethod, res)
    return res


class Rpc:
    def __init__(self, attach=False):
        import frida
        if attach:
            self._sess = frida.attach(PROC)
            pid = None
        else:
            if not os.path.exists(GAME):
                raise FileNotFoundError("game exe not found: %s" % GAME)
            pid = frida.spawn([GAME])
            self._sess = frida.attach(pid)
        self._sc = self._sess.create_script(open(GLUE, encoding="utf-8").read(), runtime="qjs")
        self._sc.load()
        self._ex = self._sc.exports_sync
        if pid is not None:
            frida.resume(pid)
        self._load_dll()

    def _load_dll(self):
        data = open(DLL, "rb").read()
        # 轮询到 AppDomain 就绪:load_bot 在未就绪时返回非 ok,重试。
        for _ in range(120):
            try:
                r = self._ex.load_bot(data)
                if r and r.get("ok"):
                    return
            except Exception:
                pass
            time.sleep(1)
        raise RuntimeError("YiXianApi.dll 加载超时(AppDomain 未就绪)")

    def call_s(self, ctype, cmethod, ints):
        return _unwrap(ctype, cmethod, self._ex.call_s(ctype, cmethod, [int(x) for x in ints]))

    def call_str(self, ctype, cmethod, s):
        return _unwrap(ctype, cmethod, self._ex.call_str(ctype, cmethod, s))

    def close(self):
        try:
            self._sc.unload(); self._sess.detach()
        except Exception:
            pass
```

```python
# -*- coding: utf-8 -*-
"""yixian_api:按命名空间调用弈仙牌游戏函数。

    from native_hud.api import connect
    api = connect()            # spawn 游戏并注入(attach=True 接已运行的)
    api.battle.skip()
    api.state.round()          # -> {"round": 7}
"""
import json
from .registry import API


class _Namespace:
    """注册表里某个 namespace 的方法集合,绑定到一个 rpc。"""
    def __init__(self, rpc, specs):
        for s in specs:
            setattr(self, s.name, self._make(rpc, s))

    @staticmethod
    def _make(rpc, s):
        def method(*args):
            if len(args) != len(s.args):
                raise TypeError("%s.%s 需要 %d 个参数,给了 %d"
                                % (s.namespace, s.name, len(s.args), len(args)))
            if s.call == "call_s":
                raw = rpc.call_s(s.ctype, s.cmethod, list(args))
            else:
                raw = rpc.call_str(s.ctype, s.cmethod, args[0] if args else "")
            return json.loads(raw) if s.ret == "json" else raw
        method.__name__ = s.name
        method.__doc__ = s.doc
        return method


class Client:
    """按注册表把 namespace 装成属性:client.battle.skip() 等。"""
    def __init__(self, rpc):
        self._rpc = rpc
        names = []
        for s in API:
            if s.namespace not in names:
                names.append(s.namespace)
        for ns in names:
            specs = [s for s in API if s.namespace == ns]
            setattr(self, ns, _Namespace(rpc, specs))

    def close(self):
        if hasattr(self._rpc, "close"):
            self._rpc.close()


def connect(attach=False):
    from ._rpc import Rpc
    return Client(Rpc(attach=attach))
```
